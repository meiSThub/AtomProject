## 对象保存到文件
``` java
/**********对象保存**********/
/**
 * 保存对象
 * @param path
 * @param obj
 * @return
 */
def saveObject(String path, def obj) {
    try {
        def file = new File(path)
        if (!file.exists()) {
            file.createNewFile()
        }
        file.withObjectOutputStream { outputStream ->
            outputStream.writeObject(obj)
        }
    } catch (Exception e) {
        e.printStackTrace()
    }
}

/**
 * 从文件中读取对象
 * @param path
 * @return
 */
def readObject(String path) {
    def obj = null
    try {
        def file = new File(path)
        if (!file.exists()) {
            return null
        }
        file.withObjectInputStream { inputStream ->
            obj = inputStream.readObject()
        }
    } catch (Exception e) {
        e.printStackTrace()
    }
    return obj
}
```
